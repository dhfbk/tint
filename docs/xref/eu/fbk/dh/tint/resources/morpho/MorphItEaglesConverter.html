131     public static void main(String[] args) {
132         final CommandLine cmd = CommandLine
133                 .parser()
134                 .withName("morphit-converter")
135                 .withHeader("Convert Morph-It dataset to be compliant with fstan")
136                 .withOption("i", "input", "input file", "FILE", CommandLine.Type.FILE_EXISTING, true, false, true)
137                 .withOption("o", "output", "output file", "FILE", CommandLine.Type.FILE, true, false, true)
138                 .withOption("f", "fstan-command", "fstan TextPro command", "COMMAND", CommandLine.Type.FILE_EXISTING,
139                         true, false, true)
140                 .withOption("m", "fstan-model", "fstan TextPro model", "MODEL FILE", CommandLine.Type.FILE_EXISTING,
141                         true, false, true)
142                 .withOption(null, "no-fstan", "Do not use fstan for initial population")
143                 .withOption(null, "use-spaces", "Use spaces (instead of tabs) as separator")
144                 .withLogger(LoggerFactory.getLogger("eu.fbk.dh")).parse(args);
145 
146         final File inputPath = cmd.getOptionValue("i", File.class);
147         final File outputPath = cmd.getOptionValue("o", File.class);
148 
149         final String fstanCommand = cmd.getOptionValue("f", String.class);
150         final String fstanModel = cmd.getOptionValue("m", String.class);
151 
152         boolean useFstan = !cmd.hasOption("no-fstan");
153         boolean useSpaces = cmd.hasOption("use-spaces");
154 
155         char separator = '\t';
156         if (useSpaces) {
157             separator = ' ';
158         }
159 
160         FstanRunner runner = new FstanRunner(fstanCommand, fstanModel);
161         HashSet<String> buffer = new HashSet<>();
162 
163         HashSet<String> allForms = new HashSet<>();
164 
165         try {
166             BufferedWriter writer = new BufferedWriter(new FileWriter(outputPath));
167             HashSet<String> types = new HashSet<>();
168 
169             HashMultimap<String, String> typeToForm = HashMultimap.create();
170             FrequencyHashSet<String> formMeanings = new FrequencyHashSet<>();
171 
172             List<String> lines = Files.readLines(inputPath, Charsets.ISO_8859_1);
173             for (String line : lines) {
174                 line = line.trim();
175                 if (line.length() == 0) {
176                     continue;
177                 }
178 
179                 String[] parts = line.split("\\s+");
180                 if (parts.length != 3) {
181                     LOGGER.error("Invalid line: {}", line);
182                     continue;
183                 }
184 
185                 String form = parts[0];
186                 String lemma = parts[1];
187                 String morpho = parts[2];
188 
189                 Matcher matcher = morphoType.matcher(morpho);
190                 if (!matcher.find()) {
191                     LOGGER.warn("Invalid pattern: {}", morpho);
192                     continue;
193                 }
194 
195                 String type = matcher.group(1);
196                 if (noLemmaTypes.containsKey(type)) {
197                     continue;
198                 }
199 
200                 allForms.add(form);
201 
202                 if (skipTypes.contains(type)) {
203                     continue;
204                 }
205                 if (manuallyMapped.keys().contains(morpho)) {
206                     continue;
207                 }
208 
209                 types.add(morpho);
210                 typeToForm.put(morpho, form);
211                 formMeanings.add(form);
212             }
213 
214             for (String form : formMeanings.keySet()) {
215                 if (formMeanings.get(form) == 1) {
216                     buffer.add(form);
217                 }
218             }
219 
220             HashMap<String, String> mappings = new HashMap<>();
221 
222             for (String type : typeToForm.keySet()) {
223 
224                 HashSet<String> nonAmbiguous = new HashSet<>();
225                 for (String form : typeToForm.get(type)) {
226                     if (buffer.contains(form)) {
227                         nonAmbiguous.add(form);
228                     }
229 
230                     if (nonAmbiguous.size() > 1000) {
231                         break;
232                     }
233                 }
234 
235                 ArrayList<String> toFstan = new ArrayList<>(nonAmbiguous);
236                 ArrayList<String[]> run = runner.run(toFstan);
237 
238                 FrequencyHashSet<String> fstanForms = new FrequencyHashSet<>();
239 
240                 for (int i = 0; i < toFstan.size(); i++) {
241                     String[] res = run.get(i);
242 
243                     if (res.length == 1) {
244                         continue;
245                     }
246 
247                     for (int j = 1; j < res.length; j++) {
248                         String okForm = res[j];
249                         Matcher matcher = fstanPattern.matcher(okForm);
250                         if (!matcher.find()) {
251                             LOGGER.error("Error in form: {}", okForm);
252                             continue;
253                         }
254 
255                         String suffix = matcher.group(2);
256                         fstanForms.add(suffix);
257                     }
258                 }
259 
260                 if (fstanForms.size() == 0) {
261                     LOGGER.warn("No forms for: {}", type);
262                     continue;
263                 }
264 
265                 String mf = fstanForms.mostFrequent();
266 
267                 mappings.put(type, mf);
268             }
269 
270             ArrayList<String> allFormsArray = new ArrayList<>(allForms);
271             HashMultimap<String, String> fstanMorpho = HashMultimap.create();
272 
273             if (useFstan) {
274                 LOGGER.info("Running fstan");
275                 ArrayList<String[]> run = runner.run(allFormsArray);
276                 for (int i = 0; i < allFormsArray.size(); i++) {
277                     String form = allFormsArray.get(i);
278                     String[] morphos = run.get(i);
279                     if (morphos.length > 1) {
280                         for (int j = 1; j < morphos.length; j++) {
281                             String morpho = morphos[j];
282                             fstanMorpho.put(form, morpho);
283                         }
284                     }
285                 }
286             }
287 
288             LOGGER.info("Adding unknown forms");
289             lines = Files.readLines(inputPath, Charsets.ISO_8859_1);
290             for (String line : lines) {
291                 line = line.trim();
292                 if (line.length() == 0) {
293                     continue;
294                 }
295 
296                 String[] parts = line.split("\\s+");
297                 if (parts.length != 3) {
298                     LOGGER.error("Invalid line: {}", line);
299                     continue;
300                 }
301 
302                 String form = parts[0];
303                 String lemma = parts[1];
304                 if (fstanMorpho.containsKey(form)) {
305                     for (String s : fstanMorpho.get(form)) {
306                         writer.append(form).append(separator)
307                                 .append(lemma).append(separator)
308                                 .append(s).append("\n");
309                     }
310                     continue;
311                 }
312 
313                 String morpho = parts[2];
314 
315                 Matcher matcher = morphoType.matcher(morpho);
316                 if (!matcher.find()) {
317                     LOGGER.warn("Invalid pattern: {}", morpho);
318                     continue;
319                 }
320 
321                 String type = matcher.group(1);
322                 if (skipTypes.contains(type)) {
323                     continue;
324                 }
325                 if (noLemmaTypes.containsKey(type)) {
326                     writer.append(form).append(separator)
327                             .append(lemma).append(separator)
328                             .append(noLemmaTypes.get(type)).append("\n");
329                     continue;
330                 }
331 
332                 String finalMorpho = null;
333 
334                 if (manuallyMapped.keys().contains(morpho)) {
335                     for (String m : manuallyMapped.get(morpho)) {
336                         finalMorpho = lemma + m;
337                     }
338                 }
339 
340                 if (finalMorpho == null) {
341                     String eaglesMorpho = mappings.get(morpho);
342                     if (eaglesMorpho == null) {
343                         LOGGER.error(morpho);
344                         System.exit(1);
345                     }
346 
347                     finalMorpho = lemma + eaglesMorpho;
348                 }
349 
350                 writer.append(form).append(separator)
351                         .append(lemma).append(separator);
352                 if (finalMorpho.contains("/")) {
353                     writer.append(lemma).append("~");
354                 }
355                 writer.append(finalMorpho).append("\n");
356             }
357 
358             writer.append("il").append(separator).append("il").append(separator).append("il+art+m+sing").append("\n");
359             writer.append("lo").append(separator).append("il").append(separator).append("il+art+m+sing").append("\n");
360             writer.append("la").append(separator).append("la").append(separator).append("la+art+f+sing").append("\n");
361             writer.append("i").append(separator).append("il").append(separator).append("il+art+m+plur").append("\n");
362             writer.append("gli").append(separator).append("il").append(separator).append("il+art+m+plur").append("\n");
363             writer.append("le").append(separator).append("la").append(separator).append("la+art+f+plur").append("\n");
364 
365             writer.append("un").append(separator).append("un").append(separator).append("un+art+m+sing").append("\n");
366             writer.append("uno").append(separator).append("un").append(separator).append("un+art+m+sing").append("\n");
367             writer.append("una").append(separator).append("una").append(separator).append("una+art+f+sing").append("\n");
368 
369             writer.close();
370         } catch (Exception e) {
371 //            logger.error(e.getMessage());
372             e.printStackTrace();
373         }
374     }
