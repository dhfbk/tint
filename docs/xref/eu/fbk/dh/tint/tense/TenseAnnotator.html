50          }

52          public CoreLabel getToken() {
53              return token;
54          }

56          public void setToken(CoreLabel token) {
57              this.token = token;
58          }
59  
60          public List<String> getCompatibleMorpho() {
61              return compatibleMorpho;
62          }
63  
64          public void setCompatibleMorpho(List<String> compatibleMorpho) {
65              this.compatibleMorpho = compatibleMorpho;
66          }
67  
68          private Set<String> getTense() {
69              Set<String> ret = new HashSet<>();
70              List<String> compatibleMorpho = getCompatibleMorpho();
71              for (String m : compatibleMorpho) {
72                  String[] parts = m.split("\\+");
73                  if (parts.length > 3) {
74                      ret.add(parts[2] + parts[3]);
75                  }
76              }
77  
78              // todo: check imp
79              if (ret.size() > 1 && ret.contains("imppres")) {
80                  ret.remove("imppres");
81              }
82  
83              return ret;
84          }

129         public Set<String> getPatterns() {
130 
131             Set<String> ret = new HashSet<>();
132 
133             StringBuffer pattern = new StringBuffer();
134             for (int i = 0; i < tokens.size(); i++) {
135                 VerbToken token = tokens.get(i);
136 
137                 if (i < tokens.size() - 1) {
138                     String lemma = token.getToken().lemma();
139                     if (lemma.equals("venire")) {
140                         lemma = "essere";
141                     }
142                     pattern.append(lemma).append("+");
143                 } else {
144                     Set<String> tenses = token.getTense();
145                     for (String tense : tenses) {
146                         ret.add(pattern.toString() + tense);
147                     }
148                 }
149             }
150 
151             return ret;
152         }
153 
154         public String getLemma() {
155             VerbToken lastToken = tokens.get(tokens.size() - 1);
156             return lastToken.getToken().lemma();
157         }

171     public TenseAnnotator(String annotatorName, Properties prop) {
172         String tvFile = prop.getProperty(annotatorName + ".transitive_verbs", null);
173         usePrefix = PropertiesUtils.getBoolean(prop.getProperty(annotatorName + ".use_prefix"), DEFAULT_USE_PREFIX);
174         String skipTagsText = prop.getProperty(annotatorName + ".skip_tags", DEFAULT_SKIP_TAGS);
175         String verbTagsText = prop.getProperty(annotatorName + ".verb_tags", DEFAULT_VERB_TAGS);
176 
177         // todo: add property for transitive verbs file list
178         // todo: load this part once
179 
180         skipTags = new ArrayList<>();
181         verbTags = new ArrayList<>();
182 
183         String[] sParts = skipTagsText.split("\\s*,\\s*");
184         for (String sPart : sParts) {
185             skipTags.add(sPart);
186         }
187 
188         String[] vParts = verbTagsText.split("\\s*,\\s*");
189         for (String vPart : vParts) {
190             verbTags.add(vPart);
191         }
192 
193         // Load transitive verbs
194         transitiveVerbs = new HashSet<>();
195         InputStream stream = null;
196         if (tvFile != null) {
197             try {
198                 stream = new FileInputStream(tvFile);
199             } catch (FileNotFoundException e) {
200                 // continue
201             }
202         }
203         if (stream == null) {
204             stream = this.getClass().getResourceAsStream("/transitiveVerbs.txt");
205         }
206 
207         try {
208             BufferedReader reader = new BufferedReader(new InputStreamReader(stream));
209             String line;
210 
211             while ((line = reader.readLine()) != null) {
212                 line = line.trim();
213                 if (line.length() > 0) {
214                     transitiveVerbs.add(line);
215                 }
216             }
217         } catch (IOException e) {
218             e.printStackTrace();
219         }

222     @Override public void annotate(Annotation annotation) {
223         if (annotation.has(CoreAnnotations.SentencesAnnotation.class)) {
224             for (CoreMap sentence : annotation.get(CoreAnnotations.SentencesAnnotation.class)) {
225 
226                 List<CoreLabel> lastVerb = new ArrayList<>();
227 
228                 List<CoreLabel> tokens = sentence.get(CoreAnnotations.TokensAnnotation.class);
229                 boolean followedByExMark = tokens.get(tokens.size() - 1).word().equals("!");
230                 boolean preceededByNot = false;
231 
232                 for (int i = 0; i < tokens.size(); i++) {
233                     CoreLabel token = tokens.get(i);
234 
235                     String pos = token.get(CoreAnnotations.PartOfSpeechAnnotation.class);
236                     String form = token.word().toLowerCase();
237                     if (noWords.contains(form)) {
238                         preceededByNot = true;
239                     }
240 
241                     if (isSatisfied(pos, verbTags)) {
242                         lastVerb.add(token);
243                         continue;
244                     }
245                     if (isSatisfied(pos, skipTags)) {
246                         continue;
247                     }
248 
249                     if (lastVerb.size() > 0) {
250                         String tense = analyzeVerb(lastVerb, followedByExMark, preceededByNot);
251                         tokens.get(i - 1).set(TenseAnnotations.TenseAnnotation.class, tense);
252                         preceededByNot = false;
253                         lastVerb = new ArrayList<>();
254                     }
255                 }
256 
257                 if (lastVerb.size() > 0) {
258                     String tense = analyzeVerb(lastVerb, followedByExMark, preceededByNot);
259                     tokens.get(tokens.size() - 1).set(TenseAnnotations.TenseAnnotation.class, tense);
260 //                    lastVerb = new ArrayList<>();
261 //                    preceededByNot = false;
262                 }
263             }
264         }

267     static Set<String> getFromMap(Set<String> needle, Map<String, String> haystack) {
268         Set<String> ret = new HashSet<>();
269         for (String s : needle) {
270             String v = haystack.get(s);
271             if (v != null) {
272                 ret.add(v);
273             }
274         }
275         return ret;

406     }

408     private List<String> getCompatibleMorpho(String morpho, String lemma) {
409         List<String> ret = new ArrayList<>();
410 
411         String[] parts = morpho.split("\\s+");
412         for (int i = 1; i < parts.length; i++) {
413             String part = parts[i];
414 
415             String[] words = part.split("/");
416             if (words.length > 1) {
417                 part = words[0];
418                 String[] t = part.split("~");
419                 if (t.length > 1) {
420                     part = t[1];
421                 }
422             }
423 
424             String[] properties = part.split("\\+");
425             if (properties[0].equalsIgnoreCase(lemma)) {
426                 ret.add(part);
427             }
428         }
429 
430         return ret;
431     }

433     private boolean isSatisfied(String pos, List<String> tags) {
434         boolean ret = false;
435         pos = pos.toLowerCase();
436 
437         if (usePrefix) {
438             for (String tag : tags) {
439                 tag = tag.toLowerCase();
440                 if (pos.startsWith(tag)) {
441                     ret = true;
442                 }
443             }
444         } else {
445             for (String tag : tags) {
446                 tag = tag.toLowerCase();
447                 if (pos.equals(tag)) {
448                     ret = true;
449                 }
450             }
451         }
452 
453         return ret;
454     }
