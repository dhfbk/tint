44      static private void add(Gson gson, JsonObject jsonObject, TypesafeMap annotation) {
45          for (Class<?> myClass : annotation.keySet()) {
46              Object o = annotation.get((Class) myClass);
47              if (o != null) {
48                  if (myClass.isAnnotationPresent(JSONLabel.class)) {
49                      JSONLabel JsonAnnotation = myClass.getAnnotation(JSONLabel.class);
50                      String name = JsonAnnotation.value();
51                      if (name != null && name.length() > 0) {
52                          jsonObject.add(JsonAnnotation.value(), gson.toJsonTree(o));
53                      }
54  
55                      Class<?>[] serializerClasses = JsonAnnotation.serializer();
56                      for (Class<?> serializerClass : serializerClasses) {
57                          if (JsonSerializer.class.isAssignableFrom(serializerClass)) {
58                              // do stuff
59                          }
60                      }
61  
62                  }
63              }
64          }
65      }

67      class SpanSerializer implements JsonSerializer<Span> {
68  
69          @Override public JsonElement serialize(Span span, Type type,
70                  JsonSerializationContext jsonSerializationContext) {
71              JsonArray jsonArray = new JsonArray();
72              jsonArray.add(span.start());
73              jsonArray.add(span.end());
74              return jsonArray;
75          }
76      }

78      class SemanticGraphSerializer implements JsonSerializer<SemanticGraph> {
79  
80          @Override public JsonElement serialize(SemanticGraph semanticGraph, Type type,
81                  JsonSerializationContext jsonSerializationContext) {
82              JsonArray jsonArray = new JsonArray();
83  
84              for (IndexedWord root : semanticGraph.getRoots()) {
85                  JsonObject object = new JsonObject();
86                  object.addProperty("dep", "ROOT");
87                  object.addProperty("governor", 0);
88                  object.addProperty("governorGloss", "ROOT");
89                  object.addProperty("dependent", root.index());
90                  object.addProperty("dependentGloss", root.word());
91                  jsonArray.add(object);
92              }
93              for (SemanticGraphEdge edge : semanticGraph.edgeListSorted()) {
94                  JsonObject object = new JsonObject();
95                  object.addProperty("dep", edge.getRelation().toString());
96                  object.addProperty("governor", edge.getGovernor().index());
97                  object.addProperty("governorGloss", edge.getGovernor().word());
98                  object.addProperty("dependent", edge.getDependent().index());
99                  object.addProperty("dependentGloss", edge.getDependent().word());
100                 jsonArray.add(object);
101             }
102             return jsonArray;
103         }
104     }

106     class RelationTripleSerializer implements JsonSerializer<RelationTriple> {
107 
108         @Override public JsonElement serialize(RelationTriple triple, Type type,
109                 JsonSerializationContext jsonSerializationContext) {
110             JsonObject ieObject = new JsonObject();
111             ieObject.addProperty("subject", triple.subjectGloss());
112             ieObject.add("subjectSpan", jsonSerializationContext.serialize(Span.fromPair(triple.subjectTokenSpan())));
113             ieObject.addProperty("relation", triple.relationGloss());
114             ieObject.add("relationSpan", jsonSerializationContext.serialize(Span.fromPair(triple.relationTokenSpan())));
115             ieObject.addProperty("object", triple.objectGloss());
116             ieObject.add("objectSpan", jsonSerializationContext.serialize(Span.fromPair(triple.objectTokenSpan())));
117             return ieObject;
118         }
119     }

134     class CorefChainSerializer implements JsonSerializer<CorefChain> {
135 
136         @Override public JsonElement serialize(CorefChain chain, Type type,
137                 JsonSerializationContext jsonSerializationContext) {
138             CorefChain.CorefMention representative = chain.getRepresentativeMention();
139             JsonArray chainArray = new JsonArray();
140             for (CorefChain.CorefMention mention : chain.getMentionsInTextualOrder()) {
141                 JsonObject mentionObj = new JsonObject();
142                 mentionObj.addProperty("id", mention.mentionID);
143                 mentionObj.addProperty("text", Sentence
144                         .listToOriginalTextString(
145                                 annotationThreadLocal.get().get(CoreAnnotations.SentencesAnnotation.class)
146                                         .get(mention.sentNum - 1)
147                                         .get(CoreAnnotations.TokensAnnotation.class)
148                                         .subList(mention.startIndex - 1,
149                                                 mention.endIndex - 1)).trim());
150                 mentionObj.add("type", jsonSerializationContext.serialize(mention.mentionType));
151                 mentionObj.add("number", jsonSerializationContext.serialize(mention.number));
152                 mentionObj.add("gender", jsonSerializationContext.serialize(mention.gender));
153                 mentionObj.add("animacy", jsonSerializationContext.serialize(mention.animacy));
154                 mentionObj.addProperty("startIndex", mention.startIndex);
155                 mentionObj.addProperty("endIndex", mention.endIndex);
156                 mentionObj.addProperty("sentNum", mention.sentNum);
157                 mentionObj.add("position", jsonSerializationContext.serialize(mention.position.elems()));
158                 mentionObj.addProperty("isRepresentativeMention", mention == representative);
159                 chainArray.add(mentionObj);
160             }
161             return chainArray;
162         }
163     }

173     @Override
174     public void print(Annotation doc, OutputStream target, Options options) throws IOException {
175 
176         if (options.pretty) {
177             gsonBuilder.setPrettyPrinting();
178         }
179         gsonBuilder.registerTypeAdapter(SemanticGraph.class, new SemanticGraphSerializer());
180         gsonBuilder.registerTypeAdapter(Span.class, new SpanSerializer());
181         gsonBuilder.registerTypeAdapter(RelationTriple.class, new RelationTripleSerializer());
182         gsonBuilder.registerTypeAdapter(Timex.class, new TimexSerializer());
183         gsonBuilder.registerTypeAdapter(CorefChain.class, new CorefChainSerializer());
184         gsonBuilder.serializeSpecialFloatingPointValues();
185         Gson gson = gsonBuilder.create();
186 
187         JsonObject jsonObject = new JsonObject();
188         jsonObject.addProperty("docId", doc.get(CoreAnnotations.DocIDAnnotation.class));
189         jsonObject.addProperty("docDate", doc.get(CoreAnnotations.DocDateAnnotation.class));
190         jsonObject.addProperty("docSourceType", doc.get(CoreAnnotations.DocSourceTypeAnnotation.class));
191         jsonObject.addProperty("docType", doc.get(CoreAnnotations.DocTypeAnnotation.class));
192         jsonObject.addProperty("author", doc.get(CoreAnnotations.AuthorAnnotation.class));
193         jsonObject.addProperty("location", doc.get(CoreAnnotations.LocationAnnotation.class));
194         if (options.includeText) {
195             jsonObject.addProperty("text", doc.get(CoreAnnotations.TextAnnotation.class));
196         }
197 
198         add(gson, jsonObject, doc);
199 
200         // Sentences
201         if (doc.get(CoreAnnotations.SentencesAnnotation.class) != null) {
202             addSentences(gson, jsonObject, doc.get(CoreAnnotations.SentencesAnnotation.class), options);
203         }
204 
205         // Add coref values
206         annotationThreadLocal.set(doc);
207         jsonObject.add("corefs", gson.toJsonTree(doc.get(CorefCoreAnnotations.CorefChainAnnotation.class)));
208 
209 //        System.out.println(gson.toJson(jsonObject));
210 
211         Writer w = new OutputStreamWriter(target);
212         w.write(gson.toJson(jsonObject));
213         w.flush();
214     }

216     private static void addSentences(Gson gson, JsonObject jsonObject, List<CoreMap> sentences,
217             Options options) {
218         JsonArray jsonSentenceArray = new JsonArray();
219         for (CoreMap sentence : sentences) {
220             JsonObject sentenceObj = new JsonObject();
221 
222             List<CoreLabel> tokens = sentence.get(CoreAnnotations.TokensAnnotation.class);
223             sentenceObj.addProperty("id", sentence.get(CoreAnnotations.SentenceIDAnnotation.class));
224             sentenceObj.addProperty("index", sentence.get(CoreAnnotations.SentenceIndexAnnotation.class));
225             sentenceObj.addProperty("line", sentence.get(CoreAnnotations.LineNumberAnnotation.class));
226             sentenceObj.addProperty("characterOffsetBegin", tokens.get(0).beginPosition());
227             sentenceObj.addProperty("characterOffsetEnd", tokens.get(tokens.size() - 1).endPosition());
228             sentenceObj.addProperty("text", sentence.get(CoreAnnotations.TextAnnotation.class));
229 
230             // Dependencies
231             sentenceObj.add("basic-dependencies",
232                     gson.toJsonTree(sentence.get(SemanticGraphCoreAnnotations.BasicDependenciesAnnotation.class)));
233             sentenceObj.add("collapsed-dependencies", gson.toJsonTree(
234                     sentence.get(SemanticGraphCoreAnnotations.CollapsedDependenciesAnnotation.class)));
235             sentenceObj.add("collapsed-ccprocessed-dependencies", gson.toJsonTree(
236                     sentence.get(SemanticGraphCoreAnnotations.CollapsedCCProcessedDependenciesAnnotation.class)));
237 
238             // Constituents
239             Tree tree = sentence.get(TreeCoreAnnotations.TreeAnnotation.class);
240             StringWriter treeStrWriter = new StringWriter();
241             TreePrint treePrinter = options.constituentTreePrinter;
242             if (treePrinter == AnnotationOutputter.DEFAULT_CONSTITUENT_TREE_PRINTER) {
243                 // note the '==' -- we're overwriting the default, but only if it was not explicitly set otherwise
244                 treePrinter = new TreePrint("oneline");
245             }
246             treePrinter.printTree(tree,
247                     new PrintWriter(treeStrWriter, true));
248             sentenceObj.addProperty("parse", treeStrWriter.toString().trim());
249 
250             // Sentiment
251             Tree sentimentTree = sentence.get(SentimentCoreAnnotations.SentimentAnnotatedTree.class);
252             if (sentimentTree != null) {
253                 int sentiment = RNNCoreAnnotations.getPredictedClass(sentimentTree);
254                 String sentimentClass = sentence.get(SentimentCoreAnnotations.SentimentClass.class);
255                 sentenceObj.addProperty("sentimentValue", Integer.toString(sentiment));
256                 sentenceObj.addProperty("sentiment", sentimentClass.replaceAll("\\s+", ""));
257             }
258 
259             // OpenIE
260             sentenceObj.add("openie", gson.toJsonTree(sentence
261                     .get(NaturalLogicAnnotations.RelationTriplesAnnotation.class)));
262 
263             // Tokens
264             if (sentence.get(CoreAnnotations.TokensAnnotation.class) != null) {
265                 addTokens(gson, sentenceObj, sentence.get(CoreAnnotations.TokensAnnotation.class));
266             }
267 
268             add(gson, sentenceObj, sentence);
269 
270             jsonSentenceArray.add(sentenceObj);
271         }
272         jsonObject.add("sentences", jsonSentenceArray);
273     }

275     private static void addTokens(Gson gson, JsonObject sentenceObj, List<CoreLabel> tokens) {
276         JsonArray jsonTokenArray = new JsonArray();
277         for (CoreLabel token : tokens) {
278             JsonObject tokenObj = new JsonObject();
279 
280             tokenObj.addProperty("index", token.index());
281             tokenObj.addProperty("word", token.word());
282             tokenObj.addProperty("originalText", token.originalText());
283             tokenObj.addProperty("lemma", token.lemma());
284             tokenObj.addProperty("characterOffsetBegin", token.beginPosition());
285             tokenObj.addProperty("characterOffsetEnd", token.endPosition());
286             tokenObj.addProperty("pos", token.tag());
287             tokenObj.addProperty("ner", token.ner());
288             tokenObj.addProperty("normalizedNER",
289                     token.get(CoreAnnotations.NormalizedNamedEntityTagAnnotation.class));
290             tokenObj.addProperty("speaker", token.get(CoreAnnotations.SpeakerAnnotation.class));
291             tokenObj.addProperty("truecase", token.get(CoreAnnotations.TrueCaseAnnotation.class));
292             tokenObj.addProperty("truecaseText", token.get(CoreAnnotations.TrueCaseTextAnnotation.class));
293             tokenObj.addProperty("before", token.get(CoreAnnotations.BeforeAnnotation.class));
294             tokenObj.addProperty("after", token.get(CoreAnnotations.AfterAnnotation.class));
295 
296             // Timex
297             tokenObj.add("timex", gson.toJsonTree(token.get(TimeAnnotations.TimexAnnotation.class)));
298 
299             add(gson, tokenObj, token);
300 
301             jsonTokenArray.add(tokenObj);
302         }
303 
304         sentenceObj.add("tokens", jsonTokenArray);
305     }

307     public static String jsonPrint(GsonBuilder gsonBuilder, Annotation annotation) throws IOException {
308         ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
309         new JSONOutputter(gsonBuilder).print(annotation, outputStream);
310         return new String(outputStream.toByteArray(), "UTF-8");
